#include <iostream>
#include <cmath> 
#include <iomanip>  // Para formatear la salida
using namespace std;

// Función factorial
double fact(int n) {
    double f = 1;
    for (int i = 1; i <= n; i++) f *= i;
    return f;
}

// Serie de Taylor para cos(x) - devuelve término por término
double taylor_cos(double x, int n_terminos) {
    double suma = 0;
    for (int n = 0; n < n_terminos; n++) {
        int exponente = 2 * n;
        double termino = pow(x, exponente) / fact(exponente);
        if (n % 2 == 0) {
            suma += termino;
        } else {
            suma -= termino;
        }
    }
    return suma;
}

// Obtener un término específico de la serie
double obtener_termino(double x, int n) {
    int exponente = 2 * n;
    double termino = pow(x, exponente) / fact(exponente);
    if (n % 2 == 0) {
        return termino;   // Término positivo
    } else {
        return -termino;  // Término negativo
    }
}

// Obtener la expresión del término
string obtener_expresion(double x, int n) {
    int exponente = 2 * n;
    if (n == 0) {
        return "1";
    } else if (n % 2 == 0) {
        return "+ x^" + to_string(exponente) + "/" + to_string(exponente) + "!";
    } else {
        return "- x^" + to_string(exponente) + "/" + to_string(exponente) + "!";
    }
}

double error(double aprox, double real) {
    return abs((real - aprox) / real) * 100;
}

int main() {
    double x = 1.2;
    double cos_real = cos(x);
    double suma_acumulada = 0;

    // Encabezados de la tabla
    cout << left << setw(8) << "N° term."
         << left << setw(30) << "Expresion"
         << left << setw(18) << "Valor numerico"
         << left << setw(18) << "Suma acumulada"
         << left << setw(15) << "Error (%)" << endl;

    for (int n = 1; n <= 4; n++) {
        // Calcular suma acumulada con n términos
        suma_acumulada = taylor_cos(x, n);
        
        // Obtener el término actual (el último agregado)
        double termino_actual = obtener_termino(x, n - 1);
        
        // Obtener expresión del término
        string expresion;
        if (n == 1) {
            expresion = "1";
        } else if ((n - 1) % 2 == 0) {
            expresion = "+ x^" + to_string(2*(n-1)) + "/" + to_string(2*(n-1)) + "!";
        } else {
            expresion = "- x^" + to_string(2*(n-1)) + "/" + to_string(2*(n-1)) + "!";
        }
        
        // Calcular error
        double err = error(suma_acumulada, cos_real);
        
        // Mostrar resultados
        cout << left << setw(8) << n
             << left << setw(30) << expresion
             << left << setw(18) << termino_actual
             << left << setw(18) << suma_acumulada
             << left << setw(15) << err << "%" << endl;
    }
    
    
    return 0;
}
